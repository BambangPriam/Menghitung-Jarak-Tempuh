# Menghitung-Jarak-Tempuh
    #include <iostream>

    using namespace std;

    int main()
    {   float S,v,t;
    cout <<"Menghitung jarak tempuh"<<endl;
    cout <<"Masukkan kecepatan (km/jam) = ";
    cin >> v;
    cout<<"Masukkan waktu tempuh (detik) = ";
    cin>> t;

    S = v*t;
    cout<< "Jarak tempuh adalah : " <<S/3600; // dibagi 3600 karena konversi dari jam ke detik

    return 0;
    }
